import { Section } from "@/components/sections/Section"

export default function AllDay({ dict }) {
  const sec = dict.cafe.sections.all_day

  return (
    <Section id="all_day" className="main-section">
      <h2 id="all_day-heading">{sec.title}</h2>
      <p>{sec.description}</p>

      {sec.subsections &&
        Object.entries(sec.subsections).map(([subId, sub]) => (
          <Section key={subId} id={subId} className="subsection">
            <h3 id={`${subId}-heading`}>{sub.title}</h3>
            <p>{sub.description}</p>
          </Section>
        ))}
    </Section>
  )
}
