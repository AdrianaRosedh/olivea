export const metadata = {}

import AllDay    from "./content/all_day.mdx"
import Padel     from "./content/padel.mdx"
import Menu      from "./content/menu.mdx"
import Community from "./content/community.mdx"
import Ambience  from "./content/ambience.mdx"
import MobileSectionTracker from "@/components/navigation/MobileSectionTracker"

export default function Cafe({ dict }) {
  const SECTION_ORDER = [
    "all_day",
    "padel",
    "menu",
    "community",
    "ambience",
  ]

  const sectionIds = SECTION_ORDER.flatMap((key) => [
    key,
    ...(dict.cafe.sections[key].subsections
      ? Object.keys(dict.cafe.sections[key].subsections)
      : []),
  ])

  return (
    <>
      {/* just the sections */}
      <AllDay    dict={dict} />
      <Padel     dict={dict} />
      <Menu      dict={dict} />
      <Community dict={dict} />
      <Ambience  dict={dict} />

      {/* mobile tracker only */}
      <MobileSectionTracker sectionIds={sectionIds} />
    </>
  )
}
