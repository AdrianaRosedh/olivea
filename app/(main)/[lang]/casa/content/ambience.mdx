import { Section } from "@/components/sections/Section"

export default function Ambience({ dict }) {
  const sec = dict.casa.sections.ambience

  return (
    <Section id="ambience" className="main-section">
      <div className="max-w-2xl text-center">
        <h2 id="ambience-heading">{sec.title}</h2>
        <p>{sec.description}</p>
      </div>

      {sec.images?.length ? (
        <div className="mt-6 grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4 w-full">
          {sec.images.map((img, idx) => (
            <div key={idx} className="bg-white shadow rounded overflow-hidden">
              <img
                src={img.src}
                alt={img.alt ?? ""}
                className="object-cover w-full h-48"
              />
            </div>
          ))}
        </div>
      ) : null}

      {sec.subsections &&
        Object.entries(sec.subsections).map(([subId, sub]) => (
          <Section key={subId} id={subId} className="subsection">
            <div className="max-w-2xl text-center">
              <h3 id={`${subId}-heading`}>{sub.title}</h3>
              <p>{sub.description}</p>
            </div>
          </Section>
        ))}
    </Section>
  )
}
