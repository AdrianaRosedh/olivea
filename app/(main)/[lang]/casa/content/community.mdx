import { Section } from "@/components/sections/Section"

export default function Community({ dict }) {
  const sec = dict.casa.sections.community

  return (
    <Section id="community" className="main-section">
      <h2 id="community-heading">{sec.title}</h2>
      <p>{sec.description}</p>

      {sec.subsections &&
        Object.entries(sec.subsections).map(([subId, sub]) => (
          <Section key={subId} id={subId} className="subsection">
            <h3 id={`${subId}-heading`}>{sub.title}</h3>
            <p>{sub.description}</p>
          </Section>
        ))}
    </Section>
  )
}
