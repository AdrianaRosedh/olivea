export const metadata = {}

import CulinaryPhilosophy  from "./content/culinary_philosophy.mdx"
import Garden              from "./content/garden.mdx"
import DiningExperiences   from "./content/dining_experiences.mdx"
import Atmosphere          from "./content/atmosphere.mdx"
import ChefTeam            from "./content/chef_team.mdx"
import Community           from "./content/community.mdx"
import MobileSectionTracker from "@/components/navigation/MobileSectionTracker"

export default function FarmToTable({ dict }) {
  const SECTION_ORDER = [
    "culinary_philosophy",
    "garden",
    "dining_experiences",
    "atmosphere",
    "chef_team",
    "community",
  ]

  const sectionIds = SECTION_ORDER.flatMap((key) => [
    key,
    ...(dict.farmtotable.sections[key].subsections
      ? Object.keys(dict.farmtotable.sections[key].subsections)
      : []),
  ])

  return (
    <>
      <CulinaryPhilosophy dict={dict} />
      <Garden             dict={dict} />
      <DiningExperiences  dict={dict} />
      <Atmosphere         dict={dict} />
      <ChefTeam           dict={dict} />
      <Community          dict={dict} />

      <MobileSectionTracker sectionIds={sectionIds} />
    </>
  )
}
